Classic ASP Objects

Full compatibility with standard Classic ASP objects.

Request Object

Access HTTP request data including query strings, form data, cookies, and server variables.
' Get query string parameter
name = Request.QueryString("name")

' Get form data
email = Request.Form("email")

' Get cookie
userId = Request.Cookies("userId")

' Get server variable
userAgent = Request.ServerVariables("HTTP_USER_AGENT")

Response Object

Control HTTP response including output, redirects, cookies, and headers. 
' Write output
Response.Write "Hello World"

' Redirect
Response.Redirect "login.asp"

' Set cookie
Response.Cookies("userId") = "12345"
Response.Cookies("userId").Expires = DateAdd("d", 30, Now())

' Set content type
Response.ContentType = "application/json"

Server Object

Server utilities and object creation.
' Create object
Set conn = Server.CreateObject("ADODB.Connection")

' Map path
physicalPath = Server.MapPath("/data/file.txt")

' Encode HTML
encoded = Server.HTMLEncode("<script>alert('xss')</script>")

' Encode URL
urlEncoded = Server.URLEncode("hello world")

Session Object

Session state management with file-based persistence.
' Set session variable
Session("username") = "john_doe"

' Get session variable
username = Session("username")

' Set timeout (minutes)
Session.Timeout = 30

' Abandon session
Session.Abandon()

Application Object

Application-wide state management.
' Set application variable
Application("SiteName") = "My Website"

' Get application variable
siteName = Application("SiteName")

' Lock for thread-safe updates
Application.Lock()
Application("VisitorCount") = Application("VisitorCount") + 1
Application.Unlock()

G3 Custom Libraries

Modern functionality for Classic ASP applications.

G3JSON - JSON Operations

Parse and stringify JSON data with ease.
Set json = Server.CreateObject("G3JSON")

' Parse JSON string
Set obj = json.Parse("{""name"": ""John"", ""age"": 30}")
Response.Write obj.name  ' Output: John

' Create and stringify object
Set data = Server.CreateObject("Scripting.Dictionary")
data.Add "name", "Jane"
data.Add "age", 25
jsonString = json.Stringify(data)

' Load from file
Set config = json.LoadFile("config.json")

G3FILES - File System Operations

Read, write, and manage files easily.
Set fs = Server.CreateObject("G3FILES")

' Read file
content = fs.Read("data.txt")

' Write file
fs.Write "output.txt", "Hello World"

' Append to file
fs.Append "log.txt", "New log entry" & vbCrLf

' Check if file exists
If fs.Exists("config.ini") Then
    ' File exists
End If

' Get file size
size = fs.Size("document.pdf")

' List files in directory
files = fs.List("/uploads")

' Delete file
fs.Delete "temp.txt"

' Create directory
fs.MkDir "newfolder"

G3HTTP - HTTP Requests

Make HTTP requests to external APIs.
Set http = Server.CreateObject("G3HTTP")

' GET request
Set result = http.Fetch("https://api.example.com/data", "GET")
Response.Write result.body

' POST request
Set headers = Server.CreateObject("Scripting.Dictionary")
headers.Add "Content-Type", "application/json"
Set result = http.Fetch("https://api.example.com/users", "POST", "{""name"":""John""}", headers)

' Check status
If result.status = 200 Then
    Response.Write "Success!"
End If

G3MAIL - Email Sending

Send emails via SMTP.
Set mail = Server.CreateObject("G3MAIL")

' Send email
mail.Send "recipient@example.com", "sender@example.com", "Subject", "Body text", False

' Send HTML email
mail.Send "recipient@example.com", "sender@example.com", "Subject", "<h1>HTML Body</h1>", True

' Standard send (uses SMTP config from .env)
mail.SendStandard "recipient@example.com", "Subject", "Body"

G3CRYPTO - Cryptography

UUID generation and password hashing.
Set crypto = Server.CreateObject("G3CRYPTO")

' Generate UUID
uuid = crypto.UUID()

' Hash password
hash = crypto.HashPassword("mypassword")

' Verify password
isValid = crypto.VerifyPassword("mypassword", hash)
If isValid Then
    Response.Write "Password correct!"
End If

G3TEMPLATE - Template Rendering

Render templates with data.
Set tpl = Server.CreateObject("G3TEMPLATE")
Set data = Server.CreateObject("Scripting.Dictionary")

data.Add "name", "John"
data.Add "title", "Welcome"

' Render template
html = tpl.Render("template.html", data)
Response.Write html

COM Object Support

ADODB.Connection

Database connections for SQL Server, MySQL, PostgreSQL, and SQLite.
Set conn = Server.CreateObject("ADODB.Connection")
conn.Open "server=localhost;database=mydb;uid=user;pwd=pass"

Set rs = conn.Execute("SELECT * FROM users")
While Not rs.EOF
    Response.Write rs("username") & "<br>"
    rs.MoveNext
Wend

rs.Close
conn.Close

MSXML2.ServerXMLHTTP

Make HTTP requests to external services.
Set http = Server.CreateObject("MSXML2.ServerXMLHTTP")
http.Open "GET", "https://api.example.com/data", False
http.Send

If http.Status = 200 Then
    Response.Write http.ResponseText
End If

Scripting.Dictionary

Key-value pair storage.
Set dict = Server.CreateObject("Scripting.Dictionary")
dict.Add "name", "John"
dict.Add "age", 30

Response.Write dict("name")  ' Output: John

' Check if key exists
If dict.Exists("age") Then
    Response.Write dict("age")
End If

Database Support

Connect to multiple database systems using ADODB.

SQLite

Set conn = Server.CreateObject("ADODB.Connection")
conn.Open "Driver=SQLite3;Database=data.db"

' In-memory database
conn.Open "Driver=SQLite3;Database=:memory:"

MySQL

Set conn = Server.CreateObject("ADODB.Connection")
conn.Open "server=localhost;port=3306;database=mydb;uid=root;pwd=password"

PostgreSQL

Set conn = Server.CreateObject("ADODB.Connection")
conn.Open "host=localhost port=5432 dbname=mydb user=postgres password=pass sslmode=disable"

MS SQL Server

Set conn = Server.CreateObject("ADODB.Connection")
conn.Open "server=localhost;database=mydb;uid=sa;pwd=password"
